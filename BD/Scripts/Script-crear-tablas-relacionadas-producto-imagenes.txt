/* Crear tabla producto */
CREATE TABLE public.producto
(
    id bigint NOT NULL PRIMARY KEY,
    nombre character varying(100),
    descripcion character varying(10000),
    color character varying(80),
    categoria character varying(80),
    talla character varying(80),
    marca character varying(100),
    precio bigint
);

ALTER TABLE IF EXISTS public.producto
    OWNER to postgres;
/* fin crear tabla producto */



/*Crear secuencia para el id auto-incremental de la tabla imagenes */
CREATE SEQUENCE giant START 111;
/* fin crear secuencia */


/*Crear tabla imagenes relacionada con producto */
CREATE TABLE public.imagenes
(
    id integer NOT NULL DEFAULT nextval('giant'::regclass) PRIMARY KEY /*Implementar secuencia */, 
    id_producto INT REFERENCES producto(id),
    ruta character varying(500)

);

ALTER TABLE IF EXISTS public.imagenes
    OWNER to postgres;
/* fin tabla imagenes relacionada */
